    initial q0
    transition q0 q1 write 1 1
    transition q1 q2 read r0 2
    transition q2 q3 check == r0 0
    transition q3 q0 noop

end

    transition q0 q1 write 1 2
    transition q1 q2 read r0 1
    transition q2 q3 check == r0 0
    transition q3 q0 noop

end
